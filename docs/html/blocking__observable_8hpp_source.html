26class blocking_observable final

36 template<constraint::subscriber_of_type<Type> TSub>
37 void subscribe(TSub&& subscriber) const noexcept
38 {
39 subscribe_impl(subscriber);
40 }
41
42 template<constraint::observer_of_type<Type> TObserver>
43 void subscribe(TObserver&& observer) const noexcept
44 {
45 return subscribe_impl(rpp::specific_subscriber<Type, std::decay_t<TObserver>>{std::forward<TObserver>(observer)});
46 }
47
48 template<typename ...Args>
49 requires (std::is_constructible_v<dynamic_subscriber<Type>, std::decay_t<Args>...> && !constraint::variadic_is_same_type<dynamic_subscriber<Type>, Args...>)
50 void subscribe(Args&&...args) const noexcept
51 {
52 subscribe_impl(rpp::make_specific_subscriber<Type>(std::forward<Args>(args)...));
53 }
54private:
55 template<constraint::subscriber_of_type<Type> TSub>
56 void subscribe_impl(TSub&& subscriber) const noexcept
57 {
58 std::promise<bool> is_success{};
59 const auto future = is_success.get_future();
60 m_original.subscribe(create_subscriber_with_state<Type>(subscriber,
61 details::forwarding_on_next{},
62 [&](const std::exception_ptr& err, const auto& sub)
63 {
64 sub.on_error(err);
65 is_success.set_value(false);
66 },
67 [&](const auto& sub)
68 {
69 sub.on_completed();
70 is_success.set_value(true);
71 }));
72 future.wait();
73 }
74
75private:
76 OriginalObservable m_original;
77};
