39template<constraint::decayed_type Type, typename OnNext, typename OnError, typename OnCompleted>
40auto make_lift_action_by_callbacks(OnNext&& on_next, OnError&& on_error, OnCompleted&& on_completed)
41{
42 return [on_next = std::forward<OnNext>(on_next),
43 on_error = std::forward<OnError>(on_error),
44 on_completed = std::forward<OnCompleted>(on_completed)](constraint::subscriber auto&& subscriber)
45 {
46 auto subscription = subscriber.get_subscription();
47 return create_subscriber_with_state<Type>(subscription,
48 std::forward<decltype(subscriber)>(subscriber),
49 on_next,
50 on_error,
51 on_completed);
52 };
53}

64template<constraint::decayed_type NewType, details::lift_fn<NewType> OperatorFn, typename FwdThis>
65auto lift_impl(OperatorFn&& op, FwdThis&& _this)
66{
67 auto action = [new_this = std::forward<FwdThis>(_this), op = std::forward<OperatorFn>(op)](auto&& subscriber)
68 {
69 new_this.subscribe(op(std::forward<decltype(subscriber)>(subscriber)));
70 };
71
72 return specific_observable<NewType, decltype(action)>(std::move(action));
73}
