22struct forwarding_on_next
23{
24 template<constraint::subscriber TSub>
25 void operator()(auto&& v, TSub&& sub) const { sub.on_next(std::forward<decltype(v)>(v)); }
26};

28struct forwarding_on_error

30 template<constraint::subscriber TSub>
31 void operator()(const std::exception_ptr& err, TSub&& sub) const {sub.on_error(err);}
32};

34struct forwarding_on_completed

36 template<constraint::subscriber TSub>

43template<constraint::decayed_type T,
44 constraint::decayed_type State,
45 std::invocable<T, State> OnNext = forwarding_on_next,
46 std::invocable<std::exception_ptr, State> OnError = forwarding_on_error,
47 std::invocable<State> OnCompleted= forwarding_on_completed>
48class state_observer final : public interface_observer<T>
49{
50public:
51 state_observer(constraint::decayed_same_as<State> auto&& state,
52 std::invocable<T, State> auto&& on_next,
53 std::invocable<std::exception_ptr, State> auto&& on_error,
54 std::invocable<State> auto&& on_completed)
55 : m_state{std::forward<decltype(state)>(state)}
56 , m_on_next{std::forward<decltype(on_next)>(on_next)}
57 , m_on_err{std::forward<decltype(on_error)>(on_error)}
58 , m_on_completed{std::forward<decltype(on_completed)>(on_completed)} {}
59
60 state_observer(const state_observer<T, State, OnNext, OnError, OnCompleted>& other) = default;
61 state_observer(state_observer<T, State, OnNext, OnError, OnCompleted>&& other) noexcept = default;
62
63 void on_next(const T& v) const override { m_on_next(v, m_state); }
64 void on_next(T&& v) const override { m_on_next(std::move(v), m_state); }
65 void on_error(const std::exception_ptr& err) const override { m_on_err(err, m_state); }
66 void on_completed() const override { m_on_completed(m_state); }
67
68private:
69 State m_state;
70 [[no_unique_address]] OnNext m_on_next;
71 [[no_unique_address]] OnError m_on_err;
72 [[no_unique_address]] OnCompleted m_on_completed;
73};
