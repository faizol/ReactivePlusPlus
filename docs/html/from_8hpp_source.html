36void iterate(const auto& iterable,
37 const schedulers::constraint::scheduler auto& scheduler,
38 const constraint::subscriber auto& subscriber)
39{
40 constexpr auto extract_iterable = []<typename T>(const T& v) -> const auto&
41 {
42 if constexpr (std::ranges::range<T>)
43 return v;
44 else
45 return *v;
46 };
47
48 if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
49 {
50 for (const auto& v : extract_iterable(iterable))
51 subscriber.on_next(utils::as_const(v));
52 subscriber.on_completed();
53 }
54 else
55 {
56 auto worker = scheduler.create_worker(subscriber.get_subscription());
57 worker.schedule([=, index = size_t{0}]() mutable-> schedulers::optional_duration
58 {
59 const auto end = std::cend(extract_iterable(iterable));
60 auto itr = std::cbegin(extract_iterable(iterable));
61
62 std::ranges::advance(itr, static_cast<int64_t>(index), end);
63
64 if (itr != end)
65 {
66 subscriber.on_next(utils::as_const(*itr));
67 if (std::next(itr) != end) // it was not last
68 {
69 ++index;
70 return schedulers::duration{}; // re-schedule this
71 }
72 }
73
74 subscriber.on_completed();
75 return {};
76 });
77 }
78}

80template<memory_model memory_model, std::ranges::range Container, typename ...Ts>
81auto pack_to_container(Ts&& ...items)
82{
83 if constexpr (memory_model == memory_model::use_stack)
84 return Container{std::forward<Ts>(items)...};
85 else
86 // raw new call to avoid extra copy/moves for items
87 return std::shared_ptr<Container>(new Container{std::forward<Ts>(items)...});
88}
