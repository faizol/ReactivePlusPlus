44template<typename ValueExtractStrategy = std::identity>
45void iterate(const auto& iterable,
46 const schedulers::constraint::scheduler auto& scheduler,
47 const constraint::subscriber auto& subscriber)
48{
49 if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
50 {
51 for (const auto& v : extract_iterable_from_packed(iterable))
52 {
53 if (subscriber.is_subscribed())
54 subscriber.on_next(utils::as_const(v));
55 else
56 return;
57 }
58 subscriber.on_completed();
59 }
60 else
61 {
62 auto worker = scheduler.create_worker(subscriber.get_subscription());
63 worker.schedule([=, index = size_t{0}]() mutable-> schedulers::optional_duration
64 {
65 if (subscriber.is_subscribed())
66 {
67 const auto& extracted_iterable = extract_iterable_from_packed(iterable);
68 const auto end = std::cend(extracted_iterable);
69 auto itr = std::cbegin(extracted_iterable);
70
71 std::ranges::advance(itr, static_cast<int64_t>(index), end);
72
73 if (itr != end)
74 {
75 subscriber.on_next(utils::as_const(*itr));
76 if (std::next(itr) != end) // it was not last
77 {
78 ++index;
79 return schedulers::duration{}; // re-schedule this
80 }
81 }
82
83 subscriber.on_completed();
84 }
85 return schedulers::optional_duration{};
86 });
87 }
88}

90template<memory_model memory_model, std::ranges::range Container, typename ...Ts>
91auto pack_to_container(Ts&& ...items)
92{
93 if constexpr (memory_model == memory_model::use_stack)
94 return Container{std::forward<Ts>(items)...};
95 else
96 // raw new call to avoid extra copy/moves for items
97 return std::shared_ptr<Container>(new Container{std::forward<Ts>(items)...});
98}

100template<memory_model memory_model, constraint::decayed_type T, typename ...Ts>
101auto pack_variadic(Ts&& ...items)
102{
103 return pack_to_container<memory_model, std::array<T, sizeof...(Ts)>>(std::forward<Ts>(items)...);
104}

204template<memory_model memory_model>
205auto from_callable(std::invocable<> auto&& callable)
206{
207 auto obs = just<memory_model>(std::forward<decltype(callable)>(callable));
208
209 if constexpr (std::same_as<std::invoke_result_t<decltype(callable)>, void>)
210 return std::move(obs).map([](const auto& fn) { fn(); return utils::none{};});
211 else
212 return std::move(obs).map([](const auto& fn) { return fn(); });
