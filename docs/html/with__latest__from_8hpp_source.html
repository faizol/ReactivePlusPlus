27template<size_t I, constraint::observable TObs>
28void with_latest_from_subscribe(const auto& state_ptr, const TObs& observable, const auto& subscriber)
29{
30 using Type = utils::extract_observable_type_t<TObs>;
31 observable.subscribe(create_subscriber_with_state<Type>(subscriber.get_subscription().make_child(),
32 subscriber,
33 [state_ptr](auto&& v, const auto&)
34 {
35 std::lock_guard lock{state_ptr->mutexes[I]};
36 std::get<I>(state_ptr->vals) = std::forward<decltype(v)>(v);
37 },
38 forwarding_on_error{},
39 [](const auto&){}));
40}

42template<size_t...I>
43void with_latest_from_subscribe_observables(std::index_sequence<I...>,

51template<constraint::observable ...TObservables>
52struct with_latest_from_state_t
53{
54 std::array<std::mutex, sizeof...(TObservables)> mutexes{};
55 std::tuple<std::optional<utils::extract_observable_type_t<TObservables>>...> vals{};
56
57 auto apply_under_lock(const auto& selector)
58 {
59 auto lock = lock_all();
60 return std::apply(selector, vals);
61 }
62
63private:
64 auto lock_all()
65 {
66 return std::apply([](auto& ...mutexes) { return std::scoped_lock{ mutexes... }; }, mutexes);
67 }
68};

70template<constraint::decayed_type Type, constraint::observable ...TObservables, std::invocable<Type, utils::extract_observable_type_t<TObservables>...> TSelector>
71auto with_latest_from_impl(TSelector&& selector, TObservables&&...observables)
72{
73 using ResultType = std::invoke_result_t<TSelector, Type, utils::extract_observable_type_t<TObservables>...>;
74
75 return [selector=std::forward<TSelector>(selector), ...observables=std::forward<TObservables>(observables)]<constraint::subscriber_of_type<ResultType> TSub>(TSub&& subscriber)
76 {
77 auto state = std::make_shared<with_latest_from_state_t<TObservables...>>();
78
79 with_latest_from_subscribe_observables(std::make_index_sequence<sizeof...(TObservables)>{}, state, subscriber, observables...);
80
81 auto on_next = [state, selector](auto&& v, const auto& sub)
82 {
83 auto result = state->apply_under_lock([&](const auto& ...args) -> std::optional<ResultType>
84 {
85 if ((args.has_value() && ...))
86 return selector(std::forward<decltype(v)>(v), args.value()...);
87 return std::nullopt;
88 });
89
90 if (result.has_value())
91 sub.on_next(std::move(result.value()));
92 };
93
94 return create_subscriber_with_state<Type>(std::forward<TSub>(subscriber),
95 std::move(on_next),
96 forwarding_on_error{},
97 forwarding_on_completed{});
98 };
99}
