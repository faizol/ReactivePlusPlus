36 worker_strategy(const rpp::composite_subscription& sub)
37 : m_state{std::make_shared<state>()}

40 }

47 private:
48 class state : public std::enable_shared_from_this<state>
49 {
50 public:
51 void defer_at(time_point time_point, std::invocable auto&& fn)
52 {
53 if (m_sub->is_subscribed())
54 m_queue.emplace(time_point, std::forward<decltype(fn)>(fn));
55 }
56
57 void init_thread(const rpp::composite_subscription& sub)
58 {
59 auto as_shared = shared_from_this();
60
61 m_thread = std::jthread{[state = as_shared](const std::stop_token& token)
62 {
63 state->data_thread(token);
64 }};
65 m_sub.reset(sub.add([state = as_shared]
66 {
67 state->m_thread.request_stop();
68
69 if (state->m_thread.get_id() != std::this_thread::get_id())
70 state->m_thread.join();
71 else
72 state->m_thread.detach();
73 }));
74 }
75
76 private:
77 void data_thread(const std::stop_token& token)
78 {
79 std::function<void()> fn{};
80 while (!token.stop_requested())
81 {
82 if (m_queue.pop_with_wait(fn, token))
83 {
84 fn();
85 fn = {};
86 }
87 }
88
89 // clear
90 m_queue.reset();
91 }
92
93 details::queue_worker_state m_queue{};
94 std::jthread m_thread{};
95 rpp::subscription_guard m_sub = rpp::subscription_base::empty();
96 };
97
98 std::shared_ptr<state> m_state{};
99 };

104 }
