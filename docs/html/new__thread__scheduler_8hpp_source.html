59class new_thread final : public details::scheduler_tag
60{
61public:
62 class worker_strategy
63 {
64 public:
65 worker_strategy(const rpp::composite_subscription& sub)
66 : m_state{std::make_shared<state>()}
67 {
68 m_state->thread = std::jthread{[state = m_state](const std::stop_token& token)
69 {
70 state->data_thread(token);
71 }};
72 m_state->sub.reset(sub.add([state = m_state]
73 {
74 state->thread.request_stop();
75
76 if (state->thread.get_id() != std::this_thread::get_id())
77 state->thread.join();
78 else
79 state->thread.detach();
80 }));
81 }
82
83 void defer_at(time_point time_point, std::invocable auto&& fn) const
84 {
85 m_state->defer_at(time_point, std::forward<decltype(fn)>(fn));
86 }
87
88 private:
89 struct state
90 {
91 void defer_at(time_point time_point, std::invocable auto&& fn)
92 {
93 if (!sub->is_subscribed())
94 return;
95
96 {
97 std::lock_guard lock{mutex};
98 queue.emplace(time_point, ++current_id, std::forward<decltype(fn)>(fn));
99 }
100 cv.notify_one();
101 }
102
103 void data_thread(const std::stop_token& token)
104 {
105 std::function<void()> fn{};
106 while (!token.stop_requested())
107 {
108 std::unique_lock lock{ mutex };
109
110 if (!cv.wait(lock, token, [&] { return !queue.empty(); }))
111 continue;
112
113 if (!cv.wait_until(lock, token, queue.top().GetTimePoint(), [&] { return !queue.empty() && queue.top().GetTimePoint() <= clock_type::now(); }))
114 continue;
115
116 fn = std::move(queue.top().ExtractFunction());
117 queue.pop();
118
119 lock.unlock();
120
121 fn();
122 fn = {};
123 }
124
125 // clear
126 std::unique_lock lock{ mutex };
127
128 queue = std::priority_queue<schedulable>{};
129 }
130
131 std::mutex mutex{};
132 std::condition_variable_any cv{};
133 std::priority_queue<schedulable> queue{};
134 size_t current_id{};
135 std::jthread thread{};
136 rpp::subscription_guard sub = rpp::subscription_base::empty();
137 };
138
139 std::shared_ptr<state> m_state{};
140 };

142 static auto create_worker(const rpp::composite_subscription& sub = composite_subscription{})
143 {
144 return worker<worker_strategy>{sub};
145 }
