32 subjects::constraint::subject_of_type<Type> Subject,

50 composite_subscription connect(const composite_subscription& subscription = composite_subscription{}) const
51 {
52 auto subscriber = m_subject.get_subscriber();
53 auto subscriber_subscription = subscriber.get_subscription();
54
55 {
56 std::lock_guard lock(m_state->mutex);
57
58 if (!m_state->sub.is_empty())
59 return subscription;
60
61 m_state->sub = subscriber_subscription.add(subscription);
62 m_original_observable.subscribe(m_state->sub, subscriber.get_observer());
63 }
64
65 subscription.add([state = m_state, subscriber_subscription]
66 {
67 auto current_sub = composite_subscription::empty();
68 {
69 std::lock_guard lock(state->mutex);
70 std::swap(current_sub, state->sub);
71 }
72 current_sub.unsubscribe();
73 subscriber_subscription.remove(current_sub);
74 });
75
76
77 return subscription;
78 }
79
80private:
81 OriginalObservable m_original_observable;
82 Subject m_subject;
83
84 struct state_t
85 {
86 std::mutex mutex{};
87 composite_subscription sub = composite_subscription::empty();
88 };
89
90 std::shared_ptr<state_t> m_state = std::make_shared<state_t>();
91};
