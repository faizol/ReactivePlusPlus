29template<constraint::decayed_type Type>
30class dynamic_observable : public interface_observable<Type, dynamic_observable<Type>>
31{
32public:
33 dynamic_observable(constraint::on_subscribe_fn<Type> auto&& on_subscribe)
34 : m_observable{std::make_shared<specific_observable<Type, std::decay_t<decltype(on_subscribe)>>>(on_subscribe)} {}
35
36 template<constraint::observable_of_type<Type> TObs>
37 requires (!std::is_same_v<std::decay_t<TObs>, dynamic_observable<Type>>)
38 dynamic_observable(TObs&& observable)
39 : m_observable{ std::make_shared<std::decay_t<TObs>>(std::forward<TObs>(observable)) } {}
40
41 dynamic_observable(const dynamic_observable<Type>&) = default;
42 dynamic_observable(dynamic_observable<Type>&&) noexcept = default;
43
44 composite_subscription subscribe(const dynamic_subscriber<Type>& subscriber) const final
45 {
46 return m_observable->subscribe(subscriber);
47 }
48
49 template<typename ...Args>
50 requires (std::is_constructible_v<dynamic_subscriber<Type>, Args...> && !constraint::variadic_is_same_type<dynamic_subscriber<Type>, Args...>)
51 composite_subscription subscribe(Args&&...args) const
52 {
53 return m_observable->subscribe(dynamic_subscriber<Type>{std::forward<Args>(args)...});
54 }
55
56 const dynamic_observable<Type>& as_dynamic() const { return *this; }
57
58private:
59 std::shared_ptr<virtual_observable<Type>> m_observable{};
60};
